```java
class Solution {
    public int[] solution(int[] num_list) {
        int odd = 0;
        int even = 0;
        
        for (int num : num_list) {
            if (num % 2 == 0)
                even++;
            else
                odd++;
        }
        
        return new int[] {even, odd};
    }
}
```

```java
class Solution {
    public static final int ODD = 1;
    public int[] solution(int[] numbers) {
        int odd=0;
        for(int number : numbers)
            if((number&ODD)==ODD)
```
