```java
class Solution {
    public int solution(int n) {
        int answer = 0;
        for (int i=1; i<n+1; i++) {
            if (n <= i*7) {
                answer++;
                break;
            }
            answer++;
        }
        return answer;
    }
}
```

```java
class Solution {
    private static final int PIZZA_SLICE = 7;
    public int solution(int n) {
        return n%PIZZA_SLICE==0?n/PIZZA_SLICE:n/PIZZA_SLICE+1;
    }
}
```
